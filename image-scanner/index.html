// script.js document.addEventListener("DOMContentLoaded", async () => { const video = document.getElementById("video"); const scanButton = document.getElementById("scanButton"); const result = document.getElementById("result"); // Ativar a câmera if (navigator.mediaDevices.getUserMedia) { navigator.mediaDevices.getUserMedia({ video: true }) .then(stream => { video.srcObject = stream; }) .catch(error => console.error("Erro ao acessar a câmera:", error)); } // Carregar modelo de reconhecimento de imagem const model = await mobilenet.load(); scanButton.addEventListener("click", async () => { const tensor = tf.browser.fromPixels(video); const predictions = await model.classify(tensor); if (predictions.length > 0) { const label = predictions[0].className; result.textContent = `Imagem reconhecida: ${label}`; if (label.includes("banana")) { window.location.href = "https://meusite.com/conteudo-especifico"; } } }); });
